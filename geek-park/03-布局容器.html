// 全局修改antd-mobile默认的主题色

+  // 修改组件库主色
+  --adm-color-primary: #fc6627;
+}

+ // 局部修改antd-mobile的主题色 -相当于限制在某个类名之下

  • 常见疑问：为什么要写两个重复的 :root？

    由于 antd-mobile 中的主题变量也是在 :root 下声明的，所以在有些情况下会由于优先级的问题无法成功覆盖。

    通过 :root:root 可以显式地让你所写内容的优先级更高一些，从而确保主题变量的成功覆盖。

补充：CSS3 变量语法 ​

目标：学习 CSS3 变量的定义和使用。

补充：:root 表示根标签，在浏览器环境中就是 html 标签。

scss
// 定义 CSS3 变量：
:root {
  --变量名: 值;
  --color-primary: pink;
}

// 使用 CSS3 变量：
.选择器 {
  样式属性: var(--变量名);
  background-color: var(--color-primary);
}

布局容器底部 TabBar ​

目标：实现一个带有底部 tab 导航栏的页面布局容器组件，当点击底部按钮后，可切换显示不同内容

实现思路：

  • 定义一个数组来存放 TabBar 相关数据，通过遍历渲染 TabBar 子项，方便统一管理。
  • 点击 TabBar 按钮后，需要实现路由地址切换。
  • 根据路由地址，高亮对应的 TabBar 子项。

操作步骤

  • 修改基本结构：src\pages\Layout\index.tsx
jsx
import styles from "./index.module.scss";
import Icon from "@/components/Icon";
import { TabBar } from "antd-mobile";

export default function Layout() {
  // 定义一个代表  `tabBar` 的数组，用于动态渲染 `TabBar` 子项
  const tabs = [
    {
      key: "/home",
      title: "首页",
      icon: (active: boolean) =>
        active ? (
          <Icon type="iconbtn_home_sel" />
        ) : (
          <Icon type="iconbtn_home" />
        ),
    },
    {
      key: "/qa",
      title: "问答",
      icon: (active: boolean) =>
        active ? <Icon type="iconbtn_qa_sel" /> : <Icon type="iconbtn_qa" />,
    },
    {
      key: "/video",
      title: "视频",
      icon: (active: boolean) =>
        active ? (
          <Icon type="iconbtn_video_sel" />
        ) : (
          <Icon type="iconbtn_video" />
        ),
    },
    {
      key: "/profile",
      title: "我的",
      icon: (active: boolean) =>
        active ? (
          <Icon type="iconbtn_mine_sel" />
        ) : (
          <Icon type="iconbtn_mine" />
        ),
    },
  ];

  return (
    <div className={styles.root}>
      {/* 区域一：点击按钮切换显示内容的区域 */}
      <div className="tab-content">
        <h3>内容</h3>
      </div>
      {/* 区域二：按钮区域，会使用固定定位显示在页面底部 */}
      <TabBar className="geek-tabbar">
        {tabs.map((item) => (
          <TabBar.Item key={item.key} icon={item.icon} title={item.title} />
        ))}
      </TabBar>
    </div>
  );
}
