cp-radio-btn- 双向绑定 #

目标：组件支持v-model双向绑定

步骤：

  1. 定义props，modelValue
  2. 定义emits，update:ModelValue事件
  3. 插值显示modelValue ，触发自定义事件update:ModelValue
  4. 绑定v-model测试

代码

diff
<script setup lang="ts">

export interface IProps {
  options: { label: string; value: string | number }[]
  // 1.  定义props，`modelValue`
+ modelValue: string | number
}
defineProps<IProps>()

// 2. 定义emits，`update:ModelValue`事件
+ interface IEmits {
+   (eventName: 'update:modelValue', value: string | number): void
+ }
+ defineEmits<IEmits>()

</script>

<template>
  <div class="cp-radio-btn">
    <a
      v-for="item in options"
      :key="item.value"
      class="item"
      href="javascript:;"
      // 3. 动态class控制选中效果
+      :class="{ active: item.value === modelValue }"
			// 4. 点击触发自定义事件
+      @click="$emit('update:modelValue', item.value)"
    >
      {{ item.label }}
    </a>
  </div>
</template>
